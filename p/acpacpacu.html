其实现在很多前端UI组件库都有表单验证这个功能也非常完善，但是最近项目中的大型自定义表单使用这些UI库内的组件会非常慢，遂用原生方法进行。

<template>
    <form ref="form">
        <input type="text" required v-model="inputValue" />
        <button @click="submit">提交</button>
    </form>
</tempalte>

<script lang="ts" setup>
import { ref } from 'vue';

const form: any = ref(null);

const submit = (e: any) => {
    e.preventDefault();
	if (form.value.reportValidity()) {
		console.log("验证通过");
	} else {
		console.log("存在未填项");
	}
}
</script>
