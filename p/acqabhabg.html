# 前言

这个样式是逛 张洪Heo 博客的时候看见的，感觉这种方式也非常的好看所以就抄了下来，因为 @vuepress/plugin-nprogress 这个插件并不能自定义 Nprogress 的配置，所以我们直接引用原生的 Nprogress 来操作。

# 实现代码

// .vuepress/enhanceApp.js
export default ({Vue, options, router, siteData, isServer}) => {
    Vue.mixin({
        mounted() {
            // nprogress 魔改 start
            import('nprogress').then((nprogress) => {
                nprogress.configure({
                    showSpinner: false,
                    template: '<div class="bar"><div class="peg" role="bar"></div></div></div>',
                    speed: 500
                })
                router.beforeEach((to, from, next) => {
                    if (to.path !== from.path) {
                        nprogress.start()
                    }
                    next()
                })
                router.afterEach(() => {
                    nprogress.done()
                })
            })
            // nprogress 魔改 end
        }
    })
};

// .vuepress/theme/styles/index.style
@keyframes gradient {
  0% {
    background-position: 0 50%
  }
  50% {
    background-position: 100% 50%
  }
  100% {
    background-position: 0 50%
  }
}

#nprogress {
  .bar {
    position: fixed;
    left: 50%;
    width: 80px;
    background: transparent;
    transform: translateX(-50%);
    height: 6px;
    margin: 0 auto;
    top: 6px;
    overflow: hidden;

    border: 1px solid #42444a;

    .peg {
      width: 100px;
      background: linear-gradient(215deg, #4584ff 0%, #cf0db9 100%);
      background-size: 200%;
      box-shadow: none;
      animation: gradient 2s ease infinite;
    }
  }
}
